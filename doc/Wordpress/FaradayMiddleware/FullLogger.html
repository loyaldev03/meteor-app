    ::Logger.new(STDOUT)

def call(env)
  info "#{env[:method]} #{env[:url].to_s}"

def on_complete(env)
  info('Status') { env[:status].to_s }
  debug('response') { dump_headers env[:response_headers] }
  debug('body') { env[:response].body }
end
