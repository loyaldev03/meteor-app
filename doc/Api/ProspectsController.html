def enroll
	response = { :message => "prospect_data_invalid", :code => '405' }
	prospect = Prospect.new(params[:prospect])
	if prospect.save!
	  response[:message] = "Prospect was successfuly saved."
    response[:code] = '000'
    response[:prospect_id] = prospect.id
  end   
  render json: response
end
